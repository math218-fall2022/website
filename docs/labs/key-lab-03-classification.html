myLDA <- function(x, mu, sd, pi){
  K <- length(mu)
  scores <- rep(NA, K)
  for(k in 1:K){
    scores[k] <- x * mu[k]/sd^2 - mu[k]^2/(2*sd^2) + log(pi[k])
  }
  return(scores)
}
